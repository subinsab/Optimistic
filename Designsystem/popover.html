Popover

Click-triggered floating panel anchored to a trigger element. Richer than a tooltip — can contain menus, forms, and interactive content. Closes on outside click or Escape.

Click-triggered Bottom-left / bottom-right / top Rich content Outside-click close Escape key close

Anatomy

Popover structure
1
Anchor — .ym-popover-anchor, position:relative
2
Trigger — any button or element inside the anchor that calls togglePop(id)
3
Panel — .ym-popover, shows on .open class, hidden via opacity + scale
4
Arrow — CSS ::before rotated square, positioned top-left or top-right
5
Header (optional) — .ym-popover-header with title and close button
6
Body — .ym-popover-body, free-form content area
7
Footer (optional) — .ym-popover-footer, right-aligned action buttons
Placement modifiers
A
Bottom-left — default, top:calc(100%+8px); left:0
B
Bottom-right — .pop-right, arrow shifts right
C
Top — .pop-top, slides up from below anchor

import { useState, useRef, useEffect } from 'react';

function Popover({ trigger, children, placement = 'bottom-left' }) {
  const [open, setOpen] = useState(false);
  const ref = useRef(null);

  useEffect(() => {
    function handleClickOutside(e) {
      if (ref.current && !ref.current.contains(e.target)) setOpen(false);
    }
    function handleEscape(e) {
      if (e.key === 'Escape') setOpen(false);
    }
    document.addEventListener('mousedown', handleClickOutside);
    document.addEventListener('keydown', handleEscape);
    return () => {
      document.removeEventListener('mousedown', handleClickOutside);
      document.removeEventListener('keydown', handleEscape);
    };
  }, []);

  const popClass = [
    'ym-popover',
    open ? 'open' : '',
    placement === 'bottom-right' ? 'pop-right' : '',
    placement === 'top' ? 'pop-top' : '',
  ].filter(Boolean).join(' ');

  return (
    <div className="ym-popover-anchor" ref={ref}>
      {React.cloneElement(trigger, {
        onClick: () => setOpen(v => !v),
        'aria-expanded': open,
        'aria-haspopup': 'true',
      })}
      <div className={popClass}>
        {children}
      </div>
    </div>
  );
}

// Usage
<Popover
  trigger={<button className="ym-btn secondary btn-m">Actions</button>}
  placement="bottom-left"
>
  <div className="ym-popover-body" style={{padding:'8px'}}>
    <div className="pop-menu-item">View details</div>
  </div>
</Popover>
