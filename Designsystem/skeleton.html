Skeleton

Placeholder loading state that mimics the shape of the content it represents. Prevents layout shift and communicates that content is being fetched.

.ym-skeleton {
  background: var(--bg-inset);
  border-radius: 6px;
  position: relative;
  overflow: hidden;
}
.ym-skeleton::after {
  content: '';
  position: absolute; inset: 0;
  background: linear-gradient(90deg, transparent, rgba(255,255,255,.15), transparent);
  animation: shimmer 1.4s infinite;
  transform: translateX(-100%);
}
@keyframes shimmer {
  0% { transform: translateX(-100%) }
  100% { transform: translateX(100%) }
}
.ym-sk-text   { height: 14px; margin-bottom: 8px; }
.ym-sk-title  { height: 20px; margin-bottom: 12px; }
.ym-sk-avatar { width: 40px; height: 40px; border-radius: 50%; }
.ym-sk-rect   { height: 140px; border-radius: 10px; }
function Skeleton({ variant = 'text', width, className }) {
  const variantClass = {
    text: 'ym-sk-text',
    title: 'ym-sk-title',
    rect: 'ym-sk-rect',
    avatar: 'ym-sk-avatar',
  }[variant];
  return (
    <div
      className={`ym-skeleton ${variantClass} ${className || ''}`}
      style={{ width }}
    />
  );
}

// Usage
<Skeleton variant="title" width="55%" />
<Skeleton variant="text" width="100%" />
<Skeleton variant="avatar" />
