function Combobox({ options, placeholder, value, onChange }) {
  const [query, setQuery] = React.useState('');
  const [open, setOpen] = React.useState(false);
  const filtered = options.filter(o => o.toLowerCase().includes(query.toLowerCase()));
  return (
    <div className="ym-combobox">
      <div className="ym-combobox-input-wrap">
        <input
          className="ym-combobox-input"
          value={query || value || ''}
          placeholder={placeholder}
          onChange={e => setQuery(e.target.value)}
          onFocus={() => setOpen(true)}
          onBlur={() => setTimeout(() => setOpen(false), 150)}
        />
      </div>
      {open && <div className="ym-combobox-list open">
        {filtered.map(o => <div key={o} className={`ym-combobox-item ${value===o?'selected':''}`} onMouseDown={() => { onChange(o); setQuery(''); }}>>{o}</div>)}
        {!filtered.length && <div className="ym-combobox-empty">No results</div>}
      </div>}
    </div>
  );
}
