function EmptyState({ illustration, title, description, action, compact }) {
  return (
    <div className={`ym-empty ${compact ? 'compact' : ''}`}>
      <div className="ym-empty-illustration">
        {illustration}
      </div>
      <div className="ym-empty-title">{title}</div>
      {description && <div className="ym-empty-desc">{description}</div>}
      {action && <button className="ym-empty-cta" onClick={action.onClick}>{action.label}</button>}
    </div>
  );
}
