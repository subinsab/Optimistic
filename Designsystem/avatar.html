Anatomy

Avatar structure
1
Container — .ym-avatar, round by default, size class sets dimensions
2
Content — photo <img>, initials text, or SVG icon; shown as fallback in order
3
Colour palette — .av-blue, .av-green, etc., set background + text for initials
4
Status dot — .av-status.online|away|busy|offline, positioned bottom-right
5
Square shape — .av-square modifier switches border-radius to size-appropriate corner radius
6
Group — .ym-avatar-group stacks avatars with negative margin and hover lift

type AvatarSize = 'xs'|'sm'|'md'|'lg'|'xl'|'2xl';
type AvatarColor = 'blue'|'green'|'orange'|'red'|'purple'|'teal';
type StatusType = 'online'|'away'|'busy'|'offline';

interface AvatarProps {
  src?: string;
  initials?: string;
  name: string;
  size?: AvatarSize;
  color?: AvatarColor;
  square?: boolean;
  status?: StatusType;
}

function Avatar({ src, initials, name, size = 'md', color = 'blue', square, status }) {
  const cls = [
    'ym-avatar',
    `av-${size}`,
    !src ? `av-${color}` : '',
    square ? 'av-square' : '',
  ].filter(Boolean).join(' ');

  return (
    <div className={cls} aria-label={name}>
      {src
        ? <img src={src} alt={name} />
        : (initials || name.split(' ').map(n => n[0]).slice(0,2).join('').toUpperCase())
      }
      {status && <span className={`av-status ${status}`} aria-label={status} />}
    </div>
  );
}

// Usage
<Avatar name="Arjun Sharma" size="lg" color="blue" status="online" />
