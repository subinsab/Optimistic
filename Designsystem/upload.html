  cursor: pointer;

  border-color: #4065C5;
  background: rgba(64,101,197,.04);

  position: absolute; inset: 0;
  opacity: 0; cursor: pointer;

  height: 3px; background: #4065C5;
  border-radius: 2px;
  transition: width 300ms ease;

function handleDragOver(e) {
  e.preventDefault();
  e.currentTarget.classList.add('drag-over');
}
function handleDragLeave(e) {
  e.currentTarget.classList.remove('drag-over');
}
function handleDrop(e, listId) {
  e.preventDefault();
  e.currentTarget.classList.remove('drag-over');
  addFiles([...e.dataTransfer.files], listId);
}
function handleFileSelect(input, listId) {
  addFiles([...input.files], listId);
  input.value = ''; // reset so same file can be re-selected
}
function addFiles(files, listId) {
  const list = document.getElementById(listId);
  files.forEach(file => {
    const item = document.createElement('div');
    item.className = 'ym-file-item';
    item.innerHTML = `
      <div class="file-content">
        <div class="file-name">${file.name}</div>
        <div class="file-size">${formatSize(file.size)}</div>
        <div class="file-progress-wrap">
          <div class="file-progress-bar" style="width:0%"></div>
        </div>
      </div>
      <button class="file-remove-btn" onclick="this.closest('.ym-file-item').remove()">×</button>`;
    list.appendChild(item);
    // Simulate upload progress
    simulateUpload(item.querySelector('.file-progress-bar'));
  });
}
function formatSize(bytes) {
  if (bytes < 1024) return bytes + ' B';
  if (bytes < 1048576) return (bytes / 1024).toFixed(1) + ' KB';
  return (bytes / 1048576).toFixed(1) + ' MB';
}
