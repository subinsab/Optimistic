import { useState } from "react";

interface TopNavigationProps {
  activeTab?: string;
  onTabChange?: (tab: string) => void;
}

const TABS = [
  "Home", "Client", "Portfolio", "Transactions", "Reports"
];

export function TopNavigation({
  activeTab = "Home",
  onTabChange
}: TopNavigationProps) {
  const [searchOpen, setSearchOpen] = useState(false);
  const [dropdownOpen, setDropdownOpen] = useState(false);

  return (
    <nav className="h-10 bg-white border-b border-[#EDEDED] px-9 flex items-center justify-between sticky top-0 z-50">

      {/* Left: tabs */}
      <div className="flex items-center gap-1 h-full">
        {TABS.map((tab) => (
          <button
            key={tab}
            onClick={() => onTabChange?.(tab)}
            className={`relative px-2 h-10 flex items-center rounded-md transition-colors duration-200
              ${activeTab !== tab ? "hover:bg-[#F9FAFB]" : ""}`}
          >
            <span className={`font-['Sofia_Pro',sans-serif] font-normal text-sm transition-colors duration-200
              ${activeTab === tab ? "text-[#FD7149]" : "text-[#475467] hover:text-[#344054]"}`}>
              {tab}
            </span>
            {/* Animated underline */}
            <span className={`absolute bottom-0 left-2 right-2 h-0.5 bg-[#FD7149] rounded-t
              transition-all duration-300 origin-center
              ${activeTab === tab ? "scale-x-100 opacity-100" : "scale-x-0 opacity-0"}`}
            />
          </button>
        ))}
      </div>

      {/* Right: search + bell + profile */}
      <div className="flex items-center gap-1">

        {/* Search */}
        {searchOpen ? (
          <div className="flex items-center gap-2 h-8 px-3 border border-[#4065C5] rounded-[6px] shadow-[0_0_3px_rgba(64,101,197,.4)] animate-in fade-in slide-in-from-right-2 duration-150">
            <SearchIcon color="#121926" />
            <input
              autoFocus
              type="text"
              placeholder="Search here"
              onBlur={() => setSearchOpen(false)}
              className="w-36 bg-transparent border-none outline-none text-[13px] font-['Sofia_Pro',sans-serif] text-[#121926] placeholder:text-[#9AA4B2]"
            />
          </div>
        ) : (
          <button
            onClick={() => setSearchOpen(true)}
            className="w-7 h-7 flex items-center justify-center rounded-md hover:bg-[#F9FAFB] transition-all duration-150 hover:scale-110"
          >
            <SearchIcon />
          </button>
        )}

        {/* Bell */}
        <div className="relative">
          <button className="w-7 h-7 flex items-center justify-center rounded-md hover:bg-[#F9FAFB] transition-all hover:scale-110">
            <BellIcon />
          </button>
          <span className="absolute -top-1 -right-1 bg-[#2C43AB] text-white text-[9px] px-[3px] py-[1px] rounded-sm border-[1.5px] border-white min-w-[14px] text-center animate-pulse">
            22
          </span>
        </div>

        {/* Profile */}
        <div className="relative">
          <button
            onClick={() => setDropdownOpen(!dropdownOpen)}
            className="flex items-center gap-1 p-1 rounded-md hover:bg-[#F9FAFB] transition-colors"
          >
            <div className="w-7 h-7 rounded-full bg-[#3F902B] text-white flex items-center justify-center text-[11px] font-medium font-['Sofia_Pro',sans-serif]">
              AS
            </div>
            <ChevronDown className={`w-3.5 h-3.5 text-[#344054] transition-transform duration-200 ${dropdownOpen ? "rotate-180" : ""}`} />
          </button>

          {dropdownOpen && (
            <>
              <div className="fixed inset-0 z-40" onClick={() => setDropdownOpen(false)} />
              <div className="absolute top-full mt-1.5 right-0 bg-white border border-[#EDEDED] rounded-[10px] shadow-[0_8px_24px_rgba(0,0,0,.1)] min-w-[160px] py-1.5 z-50 animate-in fade-in slide-in-from-top-2 duration-150">
                <button className="w-full px-4 py-2 text-left text-[13px] text-[#475467] hover:bg-[#F9FAFB] hover:text-[#344054] font-['Sofia_Pro',sans-serif] transition-colors">Profile</button>
                <button className="w-full px-4 py-2 text-left text-[13px] text-[#475467] hover:bg-[#F9FAFB] hover:text-[#344054] font-['Sofia_Pro',sans-serif] transition-colors">Settings</button>
                <hr className="my-1 border-[#EDEDED]" />
                <button className="w-full px-4 py-2 text-left text-[13px] text-[#CD3546] hover:bg-red-50 font-['Sofia_Pro',sans-serif] transition-colors">Logout</button>
              </div>
            </>
          )}
        </div>
      </div>
    </nav>
  );
}
