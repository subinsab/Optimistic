import { motion } from "motion/react";
import { useState } from "react";

// ─── Types ───
interface SideNavProps {
  activeItem?: string;
  onItemClick?: (item: string) => void;
}

// ─── Nav Items Config ───
const navItems = [
  { id: "Vista", label: "Vista", icon: <VistaIcon /> },
  { id: "Bonds", label: "Bonds", icon: <BondsIcon /> },
  { id: "MutualFunds", label: "Mutual Funds", icon: <MutualFundsIcon /> },
  { id: "FixedDeposits", label: "Fixed Deposits", icon: <FixedDepositsIcon /> },
  { id: "Insurance", label: "Insurance", icon: <InsuranceIcon /> },
];

// ─── Main Component ───
export function SideNav({ activeItem = "Vista", onItemClick }: SideNavProps) {
  const [isExpanded, setIsExpanded] = useState(false);

  return (
    <motion.nav
      initial={false}
      animate={{ width: isExpanded ? 240 : 72 }}
      transition={{ duration: 0.3, ease: [0.4, 0.0, 0.2, 1] }}
      onMouseEnter={() => setIsExpanded(true)}
      onMouseLeave={() => setIsExpanded(false)}
      className="bg-white border-r border-[#ededed] h-screen flex flex-col py-10 px-6 overflow-hidden"
    >
      {/* Logo */}
      <div className="mb-8 h-8 flex items-center overflow-hidden relative">
        <motion.div
          initial={false}
          animate={{ opacity: isExpanded ? 0 : 1 }}
          transition={{ duration: 0.3, ease: [0.4, 0.0, 0.2, 1] }}
          className="absolute"
        >
          <LogoIcon />
        </motion.div>
        <motion.div
          initial={false}
          animate={{ opacity: isExpanded ? 1 : 0 }}
          transition={{ duration: 0.3, ease: [0.4, 0.0, 0.2, 1] }}
          className="absolute"
        >
          <LogoFull />
        </motion.div>
      </div>

      {/* Nav Items */}
      <div className="flex flex-col gap-3">
        {navItems.map((item) => (
          <NavItem
            key={item.id}
            icon={item.icon}
            label={item.label}
            isExpanded={isExpanded}
            isActive={activeItem === item.id}
            onClick={() => onItemClick?.(item.id)}
          />
        ))}
      </div>
    </motion.nav>
  );
}

// ─── NavItem Sub-component ───
function NavItem({ icon, label, isExpanded, isActive, onClick }) {
  return (
    <button
      onClick={onClick}
      className={`flex items-center gap-2 h-8 rounded-[6px] cursor-pointer border-none p-0 w-full
        transition-colors duration-200
        ${isActive ? "" : "hover:bg-[#F9FAFB]"}`}
    >
      <span className="w-8 h-8 border-[0.5px] border-[#EEF2F6] rounded-[6px] flex items-center justify-center shrink-0">
        {icon}
      </span>
      <motion.span
        initial={false}
        animate={{ width: isExpanded ? "auto" : 0, opacity: isExpanded ? 1 : 0 }}
        transition={{ duration: 0.3, ease: [0.4, 0.0, 0.2, 1] }}
        className={`text-[12px] font-medium leading-[18px] whitespace-nowrap overflow-hidden
          ${isActive ? "text-[#FD7149]" : "text-[#121926]"}`}
        style={{ fontFamily: "'Inter', system-ui, sans-serif" }}
      >
        {label}
      </motion.span>
    </button>
  );
}
