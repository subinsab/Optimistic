Ghost border — 1.5px solid, matches variant colour

Loading spinner — 15px circle, 0.7s spin, replaces label during async

Disabled — 0.4 opacity + cursor: not-allowed, no hover/active state
Usage Rules
→
Max 1 Filled Primary per section. Use Ghost or text variants for secondary actions.
→
Error / Red only for destructive, irreversible actions (delete, revoke).
→
Loader state disables click + shows spinner. Restore after async resolves.
→
Icon-only buttons must have aria-label for accessibility.
→
Size pairing: use M (36px) in forms, L (44px) as page CTA, S (28px) in dense tables/toolbars.

<!-- Ghost — Blue, M -->
<button class="ym-btn btn-ghost-blue btn-m">Export</button>

<!-- Icon-only — must have aria-label -->
<button class="ym-btn btn-primary btn-icon btn-m" aria-label="Add">
  <svg width="16" height="16">...</svg>
</button>

<!-- Loader (while loading) -->
<button class="ym-btn btn-primary btn-m" disabled>
  <span class="btn-spinner"></span>
  Loading…
</button>

<!-- Disabled -->
<button class="ym-btn btn-primary btn-m" disabled>Unavailable</button>

<!-- Error / destructive -->
<button class="ym-btn btn-red btn-m">Delete Record</button>

// Async loader pattern
async function handleSubmit(btn) {
  const originalHTML = btn.innerHTML;

  // Enter loading state
  btn.disabled = true;
  btn.innerHTML = '<span class="btn-spinner"></span> Saving…';

  try {
    await saveData(); // your async call
    btn.innerHTML = '✓ Saved';
    setTimeout(() => {
      btn.innerHTML = originalHTML;
      btn.disabled = false;
    }, 1500);
  } catch (e) {
    btn.innerHTML = originalHTML;
    btn.disabled = false;
  }
}

// Tab switching (for the type tabs above the demo)
function switchTypeTab(type) {
  document.querySelectorAll('.btn-type-tab').forEach(t => t.classList.remove('active'));
  document.querySelectorAll('.btn-type-panel').forEach(p => p.classList.remove('active'));
  document.getElementById('tab-' + type).classList.add('active');
  document.getElementById('panel-' + type).classList.add('active');
}
