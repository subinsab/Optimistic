Notification

function Notification({ variant = 'info', title, message, actions, onDismiss, barStyle = false }) {
  const ICONS = { info: '...', success: '...', warning: '...', error: '...' };
  const cls = [
    'ym-notification',
    `notif-${variant}`,
    barStyle ? 'notif-bar' : '',
  ].filter(Boolean).join(' ');

  return (
    <div className={cls} role="alert">
      <span className="notif-icon" aria-hidden="true"
            dangerouslySetInnerHTML={{ __html: ICONS[variant] }} />
      <div className="notif-body">
        {title && <div className="notif-title">{title}</div>}
        <div className="notif-msg">{message}</div>
        {actions && <div className="notif-actions">{actions}</div>}
      </div>
      {onDismiss && (
        <button className="notif-close" aria-label="Dismiss" onClick={onDismiss}>
          {/* × SVG */}
        </button>
      )}
    </div>
  );
}
