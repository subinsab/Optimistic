class Stepper {
  constructor(containerSelector, steps) {
    this.el = document.querySelector(containerSelector);
    this.steps = steps; // [{ title, subtitle, content }]
    this.current = 0;
    this.render();
  }

  render() {
    this.el.innerHTML = this.steps.map((s, i) => {
      const state = i < this.current ? 'completed'
                  : i === this.current ? 'active'
                  : '';
      const nodeContent = i < this.current
        ? '<svg ...></svg>'  // check icon
        : i + 1;
      return `
        <div class="ym-step ${state}" data-step="${i}">
          <div class="step-node">${nodeContent}</div>
          <div class="step-label-wrap">
            <span class="step-title">${s.title}</span>
            ${s.subtitle ? `<span class="step-subtitle">${s.subtitle}</span>` : ''}
          </div>
        </div>`;
    }).join('');
  }

  next() { if (this.current < this.steps.length - 1) { this.current++; this.render(); } }
  back() { if (this.current > 0) { this.current--; this.render(); } }
  setError(step) { /* add .error class to step node */ }
}
