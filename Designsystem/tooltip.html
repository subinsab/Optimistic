Tooltip

Small contextual label that appears on hover or focus to provide supplementary information about an element. Pure CSS-driven with 4 placement options and a dark background.

Top / Bottom / Left / Right Hover + Focus Auto arrow CSS-only option Max 220px

Anatomy

Tooltip structure
1
Wrapper — .ym-tooltip-wrap, position:relative, display:inline-block
2
Trigger — any focusable element (button, link, span) inside the wrapper
3
Bubble — .ym-tooltip, dark background #1A1D27, border-radius 6px
4
Arrow — CSS ::after pseudo-element, 5px border triangle pointing toward trigger
5
Placement modifier — .tt-bottom, .tt-left, .tt-right on the wrapper
Trigger types
A
Icon buttons — most common; label not visible by default
B
Truncated text — reveal full text on hover
C
Disabled controls — explain why the action is unavailable
D
Info icons — supplementary definition or help text

type TooltipPlacement = 'top' | 'bottom' | 'left' | 'right';

interface TooltipProps {
  content: React.ReactNode;
  placement?: TooltipPlacement;
  children: React.ReactElement;
}

function Tooltip({ content, placement = 'top', children }: TooltipProps) {
  const modClass = placement !== 'top' ? `tt-${placement}` : '';
  return (
    <div className={`ym-tooltip-wrap ${modClass}`}>
      {children}
      <div className="ym-tooltip" role="tooltip">
        {content}
      </div>
    </div>
  );
}

// Usage
<Tooltip content="Approve this loan" placement="top">
  <button className="ym-btn primary btn-m">Approve</button>
</Tooltip>
