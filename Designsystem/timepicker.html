let tpHour = 9, tpMin = 0, tpAmPm = 'AM';

function buildDrum(listEl, count, start, selected) {
  listEl.innerHTML = '';
  for (let i = start; i <= count; i++) {
    const el = document.createElement('div');
    el.className = 'tp-drum-item' + (i === selected ? ' selected' : '');
    el.textContent = String(i).padStart(2, '0');
    el.dataset.val = i;
    el.addEventListener('click', () => {
      if (listEl.id === 'tpHourList') tpHour = i;
      else tpMin = i;
      refreshDrums();
    });
    listEl.appendChild(el);
  }
  // scroll selected item to centre
  const idx = selected - start;
  listEl.scrollTop = idx * 40 - 60;
}

function refreshDrums() {
  buildDrum(document.getElementById('tpHourList'), 12, 1, tpHour);
  buildDrum(document.getElementById('tpMinList'),  59, 0, tpMin);
}

function drumStep(col, dir) {
  if (col === 'hour') { tpHour = ((tpHour - 1 + dir + 12) % 12) + 1; }
  else               { tpMin  = (tpMin + dir + 60) % 60; }
  refreshDrums();
}

function setAmPm(val) {
  tpAmPm = val;
  document.getElementById('tpAM').classList.toggle('active', val === 'AM');
  document.getElementById('tpPM').classList.toggle('active', val === 'PM');
}

function toggleTimePicker() {
  const panel = document.getElementById('tpPanel');
  const isOpen = panel.classList.contains('open');
  if (!isOpen) { refreshDrums(); }
  panel.classList.toggle('open');
}

function applyTime() {
  document.getElementById('tpInput').value =
    String(tpHour).padStart(2,'0') + ' : ' + String(tpMin).padStart(2,'0') + ' ' + tpAmPm;
  document.getElementById('tpPanel').classList.remove('open');
}

function clearTime() {
  document.getElementById('tpInput').value = '';
  document.getElementById('tpPanel').classList.remove('open');
}

document.addEventListener('click', e => {
  if (!e.target.closest('.ym-timepicker-wrap'))
    document.querySelectorAll('.ym-time-panel.open').forEach(p => p.classList.remove('open'));
});
