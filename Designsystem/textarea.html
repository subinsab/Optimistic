.ym-ta-wrap { display: flex; flex-direction: column; gap: 5px; } .ym-ta-label { font-size: 13px; font-weight: 600; color: var(--text-1); } .ym-ta-field-wrap { position: relative; } .ym-ta { font-family: 'Inter', sans-serif; font-size: 14px; color: var(--text-1); background: var(--bg-panel); border: 1.5px solid var(--border); border-radius: 8px; width: 100%; outline: none; padding: 10px 12px; line-height: 1.6; resize: vertical; min-height: 88px; transition: border-color 150ms, box-shadow 150ms; } .ym-ta::placeholder { color: var(--text-3); } .ym-ta:focus { border-color: #4065C5; box-shadow: 0 0 0 3px rgba(64,101,197,.12); } .ym-ta:disabled { opacity: 0.45; cursor: not-allowed; background: var(--bg-inset); resize: none; } .ym-ta.ta-error { border-color: #CD3546; } .ym-ta.ta-success { border-color: #3F902B; } .ym-ta.ta-no-resize { resize: none; } .ta-counter { position: absolute; bottom: 8px; right: 10px; font-size: 11px; color: var(--text-3); font-family: monospace; pointer-events: none; } .ta-counter.over { color: #CD3546; } .ym-ta-hint { font-size: 11.5px; color: var(--text-3); } .ym-ta-hint.hint-error { color: #CD3546; } .ym-ta-hint.hint-success { color: #3F902B; }
import React, { useState } from 'react'; interface TextareaProps extends React.TextareaHTMLAttributes<HTMLTextAreaElement> { label?: string; hint?: string; state?: 'default' | 'error' | 'success'; required?: boolean; maxLength?: number; showCounter?: boolean; autoGrow?: boolean; } export const Textarea: React.FC<TextareaProps> = ({ label, hint, state = 'default', required, maxLength, showCounter, autoGrow, onChange, ...props }) => { const [count, setCount] = useState(0); const stateClass = state === 'error' ? 'ta-error' : state === 'success' ? 'ta-success' : ''; const handleChange = (e: React.ChangeEvent<HTMLTextAreaElement>) => { setCount(e.target.value.length); if (autoGrow) { e.target.style.height = 'auto'; e.target.style.height = e.target.scrollHeight + 'px'; } onChange?.(e); }; return ( <div className="ym-ta-wrap"> {label && <label className="ym-ta-label">{label}{required && <span className="req">*</span>}</label>} <div className="ym-ta-field-wrap"> <textarea className={`ym-ta ${stateClass} ${showCounter ? 'ta-no-resize' : ''}`} maxLength={maxLength} onChange={handleChange} style={showCounter ? { paddingBottom: '28px' } : undefined} {...props} /> {showCounter && maxLength && ( <span className={`ta-counter ${count > maxLength * 0.9 ? 'over' : ''}`}> {count} / {maxLength} </span> )} </div> {hint && <span className={`ym-ta-hint hint-${state}`}>{hint}</span>} </div> ); };
