interface LinkProps {
  href?: string;
  variant?: 'default' | 'subtle' | 'danger';
  size?: 'sm' | 'md' | 'lg';
  external?: boolean;
  disabled?: boolean;
  onClick?: () => void;
  children: React.ReactNode;
}

function Link({ href, variant = 'default', size = 'md', external, disabled, onClick, children }: LinkProps) {
  const cls = [
    'ym-link',
    variant !== 'default' ? `link-${variant}` : '',
    size !== 'md' ? `link-${size}` : '',
    external ? 'ym-link-ext-icon' : '',
    disabled ? 'link-disabled' : '',
  ].filter(Boolean).join(' ');

  if (!href) {
    return <button className={cls} onClick={onClick} disabled={disabled}>{children}</button>;
  }
  return (
    <a className={cls} href={href}
       target={external ? '_blank' : undefined}
       rel={external ? 'noopener noreferrer' : undefined}
       aria-disabled={disabled}>
      {children}
      {external && <ExternalIcon />}
    </a>
  );
}
