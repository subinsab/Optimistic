.ym-choicebox {
  display: flex; gap: 14px; padding: 16px;
  border: 1.5px solid var(--border); border-radius: 10px;
  cursor: pointer; background: var(--bg-panel);
  transition: border-color .15s, background .15s;
}
.ym-choicebox.selected {
  border-color: var(--accent);
  background: var(--accent-dim);
}
function ChoiceboxGroup({ options, value, onChange }) {
  return (
    <div className="ym-choicebox-group">
      {options.map(opt => (
        <div
          key={opt.value}
          className={`ym-choicebox ${value === opt.value ? 'selected' : ''}`}
          onClick={() => onChange(opt.value)}
        >
          <div className="ym-choicebox-indicator">
            <div className="ym-choicebox-dot" />
          </div>
          <div className="ym-choicebox-content">
            <div className="ym-choicebox-label">{opt.label}</div>
            {opt.desc && <div className="ym-choicebox-desc">{opt.desc}</div>}
          </div>
        </div>
      ))}
    </div>
  );
}
