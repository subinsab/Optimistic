Anatomy

Badge structure
1
Container — .ym-badge, inline-flex, pill shape (border-radius 20px)
2
Variant modifier — e.g. .badge-success, sets background, text, and border colors
3
Optional dot — .badge-dot adds a 6×6px filled circle via ::before
4
Optional icon — SVG inside the badge, sits before the label text
5
Count overlay — .ym-badge-count on a position:relative wrapper
Variant guide
D
Default — neutral label, no semantic meaning
I
Info — informational state, verified, confirmed
S
Success — active, disbursed, approved
W
Warning — pending, under review, requires action
E
Error — rejected, overdue, failed
A
Accent — new, featured, highlighted

type BadgeVariant = 'default' | 'info' | 'success' | 'warning' | 'error' | 'accent';
type BadgeSize = 'sm' | 'md' | 'lg';

interface BadgeProps {
  variant?: BadgeVariant;
  solid?: boolean;
  dot?: boolean;
  size?: BadgeSize;
  children: React.ReactNode;
}

function Badge({ variant = 'default', solid = false, dot = false, size = 'md', children }) {
  const cls = [
    'ym-badge',
    solid ? `badge-${variant}-solid` : `badge-${variant}`,
    dot ? 'badge-dot' : '',
    size === 'sm' ? 'badge-sm' : size === 'lg' ? 'badge-lg' : '',
  ].filter(Boolean).join(' ');
  return <span className={cls}>{children}</span>;
}

// Usage
<Badge variant="success">Active</Badge>
<Badge variant="error" solid>Overdue</Badge>
<Badge variant="warning" dot size="sm">Pending</Badge>
