3
Chevron icon — 14px, rotates 180° when open. Color `--text-3`.
4
Menu — floats 4px below trigger. 10px radius, `--bg-panel`, border `--border`. Max 240px height, scrollable.

Menu Item — 9px 12px padding, 14px font. Hover: `--bg-inset`. Selected: accent color + dot.
6
Section Label (optional) — 10px uppercase, `--text-3`. Groups items visually.

<!-- Dropdown select --> <div class="ym-dd-wrap"> <label class="ym-dd-label">Loan Type</label> <div class="ym-dd-trigger" onclick="toggleDD('dd1')"> <span class="ym-dd-placeholder" id="dd1-val">Select type…</span> <svg class="ym-dd-chevron" width="14" height="14" viewBox="0 0 14 14" fill="none"> <path d="M3 5l4 4 4-4" stroke="currentColor" stroke-width="1.5" stroke-linecap="round"/></svg> </div> <div class="ym-dd-menu" id="dd1"> <div class="ym-dd-item" onclick="selectDD('dd1','Secured')">Secured Loan</div> <div class="ym-dd-item" onclick="selectDD('dd1','Unsecured')">Unsecured Loan</div> <div class="ym-dd-item disabled">Unavailable Option</div> </div> </div>
.ym-dd-wrap { position: relative; display: flex; flex-direction: column; gap: 5px; } .ym-dd-trigger { display: flex; align-items: center; justify-content: space-between; height: 36px; padding: 0 12px; background: var(--bg-panel); border: 1.5px solid var(--border); border-radius: 8px; cursor: pointer; font-size: 14px; transition: border-color 150ms, box-shadow 150ms; } .ym-dd-trigger.dd-open { border-color: #4065C5; box-shadow: 0 0 0 3px rgba(64,101,197,.12); } .ym-dd-chevron { transition: transform 200ms; } .ym-dd-trigger.dd-open .ym-dd-chevron { transform: rotate(180deg); } .ym-dd-menu { position: absolute; top: calc(100% + 4px); left: 0; right: 0; z-index: 200; background: var(--bg-panel); border: 1px solid var(--border); border-radius: 10px; box-shadow: 0 8px 24px rgba(0,0,0,.1); display: none; max-height: 240px; overflow-y: auto; } .ym-dd-menu.open { display: block; } .ym-dd-item { padding: 9px 12px; font-size: 14px; cursor: pointer; transition: background 100ms; } .ym-dd-item:hover { background: var(--bg-inset); } .ym-dd-item.selected { color: #4065C5; font-weight: 500; } .ym-dd-item.disabled { opacity: 0.4; pointer-events: none; }
import React, { useState, useRef, useEffect } from 'react'; interface DropdownOption { value: string; label: string; disabled?: boolean; group?: string; } interface DropdownProps { label?: string; placeholder?: string; options: DropdownOption[]; value?: string; size?: 'l' | 'm' | 's'; error?: boolean; disabled?: boolean; hint?: string; onChange?: (value: string) => void; } export const Dropdown: React.FC<DropdownProps> = ({ label, placeholder = 'Select…', options, value, size = 'm', error, disabled, hint, onChange }) => { const [open, setOpen] = useState(false); const ref = useRef<HTMLDivElement>(null); const selected = options.find(o => o.value === value); useEffect(() => { const handler = (e: MouseEvent) => { if (!ref.current?.contains(e.target as Node)) setOpen(false); }; document.addEventListener('mousedown', handler); return () => document.removeEventListener('mousedown', handler); }, []); return ( <div className="ym-dd-wrap" ref={ref}> {label && <label className="ym-dd-label">{label}</label>} <div className={`ym-dd-trigger dd-${size} ${open ? 'dd-open' : ''} ${error ? 'dd-error' : ''} ${disabled ? 'dd-disabled' : ''}`} onClick={() => !disabled && setOpen(!open)} > <span className={selected ? 'ym-dd-value' : 'ym-dd-placeholder'}>{selected?.label || placeholder}</span> <svg className="ym-dd-chevron" ...>...</svg> </div> {open && ( <div className="ym-dd-menu open"> {options.map(opt => ( <div key={opt.value} className={`ym-dd-item ${opt.value === value ? 'selected' : ''} ${opt.disabled ? 'disabled' : ''}`} onClick={() => { onChange?.(opt.value); setOpen(false); }} >{opt.label}</div> ))} </div> )} {hint && <span className={`ym-dd-hint ${error ? 'hint-error' : ''}`}>{hint}</span>} </div> ); };
