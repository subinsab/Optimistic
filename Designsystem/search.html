const RESULTS = [
  { title: 'Working Capital Loan', sub: 'Loan type' },
  { title: 'Invoice Discounting',  sub: 'Product category' },
  { title: 'Aditya Birla Finance',  sub: 'Lender' },
  { title: 'Amit Kumar',            sub: 'Borrower' },
];

function handleSearchInput(input) {
  const query = input.value.trim().toLowerCase();
  const dropdown = document.getElementById('searchDropdown');
  if (!query) { dropdown.innerHTML = ''; dropdown.classList.remove('open'); return; }
  const filtered = RESULTS.filter(r => r.title.toLowerCase().includes(query));
  if (!filtered.length) {
    dropdown.innerHTML = `<div class="search-no-results">No results for "${query}"</div>`;
  } else {
    dropdown.innerHTML = filtered.map(r => `
      <div class="search-dropdown-item" onclick="selectResult('${r.title}')">
        <div class="item-label">
          <div class="item-title">${highlight(r.title, query)}</div>
          <div class="item-sub">${r.sub}</div>
        </div>
      </div>`).join('');
  }
  dropdown.classList.add('open');
}

function highlight(text, query) {
  const re = new RegExp(`(${query})`, 'gi');
  return text.replace(re, '<mark class="search-highlight">$1</mark>');
}

function clearSearch(id) {
  const input = document.getElementById(id);
  input.value = '';
  input.focus();
}

// Keyboard navigation (↑↓ highlight, Esc close)
function handleSearchKey(e) {
  const items = [...document.querySelectorAll('.search-dropdown-item')];
  const cur = items.findIndex(i => i.classList.contains('highlighted'));
  if (e.key === 'ArrowDown') {
    e.preventDefault();
    items[cur]?.classList.remove('highlighted');
    items[Math.min(cur + 1, items.length - 1)]?.classList.add('highlighted');
  } else if (e.key === 'ArrowUp') {
    e.preventDefault();
    items[cur]?.classList.remove('highlighted');
    items[Math.max(cur - 1, 0)]?.classList.add('highlighted');
  } else if (e.key === 'Escape') {
    document.getElementById('searchDropdown').classList.remove('open');
  }
}
