/* Trigger */ .ym-dp-input { height: 36px; padding: 0 36px 0 12px; background: var(--bg-panel); border: 1.5px solid var(--border); border-radius: 8px; cursor: pointer; outline: none; transition: border-color 150ms, box-shadow 150ms; } .ym-dp-input.dp-open { border-color: #4065C5; box-shadow: 0 0 0 3px rgba(64,101,197,.14); } .ym-dp-input.dp-error { border-color: #CD3546; } /* Calendar panel */ .ym-cal-panel { position: absolute; top: calc(100% + 6px); left: 0; background: var(--bg-panel); border: 1px solid var(--border); border-radius: 12px; padding: 16px; z-index: 500; box-shadow: 0 8px 32px rgba(0,0,0,.12); display: none; } .ym-cal-panel.open { display: block; } /* Day states */ .cal-sel { background: #4065C5; color: #fff; font-weight: 700; } .cal-sel-start { background: #4065C5; color: #fff; border-radius: 7px 0 0 7px; } .cal-sel-end { background: #4065C5; color: #fff; border-radius: 0 7px 7px 0; } .cal-sel-mid { background: rgba(64,101,197,.1); color: #4065C5; border-radius: 0; } .cal-today::after { content: ''; position: absolute; bottom: 3px; left: 50%; transform: translateX(-50%); width: 4px; height: 4px; border-radius: 50%; background: #4065C5; }
// State let singleView = new Date(), singlePending = null, singleSelected = null; let rangeView = new Date(), rangeStart = null, rangeEnd = null, rangePicking = false; function toggleSingleCal() { const panel = document.getElementById('single-cal'); const isOpen = panel.classList.contains('open'); closeAllCals(); if (!isOpen) { renderSingleCal(); panel.classList.add('open'); } } function renderSingleCal() { const yr = singleView.getFullYear(), mo = singleView.getMonth(); const first = new Date(yr, mo, 1).getDay(); const days = new Date(yr, mo + 1, 0).getDate(); const today = new Date(); const grid = document.getElementById('single-cal-grid'); grid.innerHTML = ''; for (let i = 0; i < first; i++) addEmptyCell(grid); for (let d = 1; d <= days; d++) { const date = new Date(yr, mo, d); const el = createDayCell(d, date, today, singlePending); el.addEventListener('click', () => { singlePending = date; renderSingleCal(); }); grid.appendChild(el); } } function singleApply() { if (singlePending) { singleSelected = singlePending; document.getElementById('dp-main').value = fmtDate(singleSelected); } closeAllCals(); } // Range picking function pickRangeDay(date) { if (!rangePicking || (rangeStart && date < rangeStart)) { rangeStart = date; rangeEnd = null; rangePicking = true; } else { rangeEnd = date; rangePicking = false; } renderRangeCal(); } function fmtDate(d) { return d.toLocaleDateString('en-GB').replace(/\//g, ' / '); }
