// Select a radio pill
function selectRadio(label) {
  const group = label.closest('.fp-radio-group');
  group.querySelectorAll('.fp-radio-pill').forEach(p => p.classList.remove('selected'));
  label.classList.add('selected');
  const input = label.querySelector('input[type="radio"]');
  if (input) input.checked = true;
}

// Back / Continue navigation
function fpBack() {
  // Navigate to previous step or page
  console.log('Navigate back');
}
function fpContinue() {
  // Validate form then navigate forward
  const name = document.getElementById('fp-name');
  if (!name.value.trim()) {
    name.style.borderColor = '#EF4444';
    name.focus();
    return;
  }
  name.style.borderColor = '';
  console.log('Navigate to step 2');
}
