// OTP auto-advance function otpNext(input) { input.value = input.value.replace(/[^0-9]/g, ''); if (input.value && input.nextElementSibling?.classList.contains('otp-input')) { input.nextElementSibling.focus(); } } function otpBack(e, input) { if (e.key === 'Backspace' && !input.value) { const prev = input.previousElementSibling; if (prev?.classList.contains('otp-input')) prev.focus(); } } // Password toggle function togglePwd(inputId, iconId) { const inp = document.getElementById(inputId); inp.type = inp.type === 'password' ? 'text' : 'password'; } // Clear button function toggleClear(input, btnId) { document.getElementById(btnId).style.display = input.value ? 'flex' : 'none'; } function clearInput(inputId, btnId) { document.getElementById(inputId).value = ''; document.getElementById(btnId).style.display = 'none'; } // Character count function updateCharCount(input, counterId, max) { const el = document.getElementById(counterId); const len = input.value.length; el.textContent = len + '/' + max; el.className = 'char-counter' + (len >= max ? ' over' : len >= max * 0.9 ? ' warn' : ''); }
