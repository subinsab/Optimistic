function Spinner({ size = 'md', color = 'accent', label = 'Loading' }) {
  return <div className={`ym-spinner sp-${size} sp-${color}`} role="status" aria-label={label} />;
}

function Dots({ color }) {
  return (
    <div className="ym-dots" role="status" aria-label="Loading">
      <span style={color ? { background: color } : undefined} />
      <span style={color ? { background: color } : undefined} />
      <span style={color ? { background: color } : undefined} />
    </div>
  );
}

function ProgressBar({ value, indeterminate = false, color }) {
  return (
    <div
      className={`ym-progress-bar${indeterminate ? ' indeterminate' : ''}`}
      role="progressbar"
      aria-valuenow={indeterminate ? undefined : value}
      aria-valuemin="0" aria-valuemax="100"
      aria-busy={indeterminate}
    >
      <div className="ym-progress-fill" style={{ width: `${value}%`, background: color }} />
    </div>
  );
}

function Skeleton({ height, width, radius = '6px' }) {
  return <div className="ym-skeleton" aria-hidden="true" style={{ height, width, borderRadius: radius }} />;
}
