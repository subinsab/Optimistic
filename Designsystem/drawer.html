.ym-drawer-overlay {
  position: fixed; inset: 0; background: rgba(0,0,0,.4);
  z-index: 1000; opacity: 0; pointer-events: none;
  transition: opacity 250ms ease;
}
.ym-drawer-overlay.open { opacity: 1; pointer-events: all; }

.ym-drawer {
  position: fixed; top: 0; bottom: 0; right: 0;
  width: 400px; max-width: 90vw;
  background: var(--bg-panel); border-left: 1px solid var(--border);

  transform: translateX(100%);
  transition: transform 280ms cubic-bezier(.4,0,.2,1);

function Drawer({ open, onClose, title, subtitle, children, footer, side = 'right', size = 'md' }) {
  useEffect(() => {
    const h = (e) => e.key === 'Escape' && onClose();
    if (open) document.addEventListener('keydown', h);
    return () => document.removeEventListener('keydown', h);
  }, [open]);

  const drawerCls = [
    'ym-drawer',
    side === 'left' ? 'ym-drawer-left' : '',
    size === 'sm' ? 'ym-drawer-sm' : size === 'lg' ? 'ym-drawer-lg' : '',
  ].filter(Boolean).join(' ');

  return (
    <div className={`ym-drawer-overlay${open ? ' open' : ''}`}
         onClick={e => e.target === e.currentTarget && onClose()}>
      <div className={drawerCls} role="dialog" aria-modal="true">
        <div className="ym-drawer-header">
          <div>
            <div className="ym-drawer-title">{title}</div>
            {subtitle && <div className="ym-drawer-subtitle">{subtitle}</div>}
          </div>
          <button className="ym-drawer-close" onClick={onClose}>×</button>
        </div>
        <div className="ym-drawer-body">{children}</div>
        {footer && <div className="ym-drawer-footer">{footer}</div>}
      </div>
    </div>
  );
}
