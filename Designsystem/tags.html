Tags

User-applied labels for categorising and filtering content. Removable, coloured, and composable in a tag input field. Square border-radius distinguishes them from pill-shaped Badges.

import { useState } from 'react';

type TagColor = 'blue'|'green'|'orange'|'red'|'purple'|'teal'|'yellow';

function Tag({ label, color, onRemove }: { label: string; color?: TagColor; onRemove?: () => void }) {
  return (
    <span className={`ym-tag${color ? ` tag-${color}` : ''}${onRemove ? ' tag-removable' : ''}`}>
      {label}
      {onRemove && (
        <button className="ym-tag-remove" onClick={onRemove} aria-label={`Remove ${label}`}>
          ×
        </button>
      )}
    </span>
  );
}

function TagInput({ value, onChange }: { value: string[]; onChange: (v: string[]) => void }) {
  const [input, setInput] = useState('');
  function add(tag: string) {
    const t = tag.trim().replace(/,$/, '');
    if (t && !value.includes(t)) onChange([...value, t]);
    setInput('');
  }
  return (
    <div className="ym-tag-input">
      {value.map(t => <Tag key={t} label={t} color="blue" onRemove={() => onChange(value.filter(v => v !== t))} />)}
      <input
        value={input}
        onChange={e => setInput(e.target.value)}
        onKeyDown={e => {
          if (e.key === 'Enter' || e.key === ',') { e.preventDefault(); add(input); }
          if (e.key === 'Backspace' && !input) onChange(value.slice(0, -1));
        }}
        placeholder="Add a tag…"
      />
    </div>
  );
}
